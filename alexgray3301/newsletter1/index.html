Let’s learn how to write a simple password encrypter together!

  1. Write your hash function, this scrambles your password text into a single larger number.

    // Mathy numbers known to scramble numbers well
    #define FNV_PRIME_64 ((1ULL<<40)+(1<<8)+0xb3)
    #define FNV_OFFSET_BASIS_64 (14695981039346656037ULL)

    // FNV-1a Hash Funtion
    uint64_t fnvHash(size_t n, const unsigned char * bytes) {
    // Start with offset
    uint64_t h = FNV_OFFSET_BASIS_64;
    // Bitwise XOR each character and multiply by a “nice” prime number
    for (size_t i = 0; i < n; i++) {
    h = (h ^ bytes[i]) * FNV_PRIME_64;
    }
    // Return result when done!
    return h;
    }
