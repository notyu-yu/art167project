import sys<br>
<br>
filename = sys.argv[1]<br>
<br>
with open(filename) as input_file:<br>
&emsp;# Replace special characters with their escape character<br>
&emsp;formatted_string = input_file.read().replace('\t', '&emsp').replace('&emsp;', '&emsp;').replace('&lt', '&lt').replace('&gt', '&gt').replace('\n', '&ltbr&gt\n')<br>
&emsp;# Write the result to another file<br>
&emsp;with open(filename+'.txt', 'w+') as output_file:<br>
&emsp;&emsp;output_file.write(formatted_string)
